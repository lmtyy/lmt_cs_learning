### 第一题：一元多项式相加 (静态顺序表)

**前置假设：** 我们假设传入的两个多项式，其内部的项**已经按照指数 (`expn`) 从大到小降序排列**。这是使用双指针法的前提。

C++

```c++
#include <iostream>
using namespace std;

#define MAX_TERMS 100

// 修复了 PPT 中的笔误，统一使用全小写 termtype
typedef struct {
    float coef; // 系数 (Coefficient)
    int expn;   // 指数 (Exponent)
} termtype;

typedef struct {
    int noelem;                 // 当前多项式包含的项数
    termtype elem[MAX_TERMS];   // 存放每一项的静态数组
} polynomial;

// 核心函数：多项式相加 Pa = Pa + Pb
void AddPolyn(polynomial &Pa, polynomial &Pb) {
    polynomial Pc;  // 准备一个临时多项式 Pc 存放结果
    Pc.noelem = 0;
    
    int i = 0; // Pa 的游标
    int j = 0; // Pb 的游标
    int k = 0; // Pc 的游标

    // 1. 双指针遍历比较
    while (i < Pa.noelem && j < Pb.noelem) {
        // 情况 A: 指数相等，同类项合并
        if (Pa.elem[i].expn == Pb.elem[j].expn) {
            float sum = Pa.elem[i].coef + Pb.elem[j].coef;
            // 系数相加不为 0 才存入结果中
            if (sum != 0.0) {
                Pc.elem[k].coef = sum;
                Pc.elem[k].expn = Pa.elem[i].expn;
                k++;
            }
            i++; 
            j++;
        } 
        // 情况 B: Pa 的指数更大 (假设按降序排)，把 Pa 的项直接放入结果
        else if (Pa.elem[i].expn > Pb.elem[j].expn) {
            Pc.elem[k] = Pa.elem[i];
            k++; 
            i++;
        } 
        // 情况 C: Pb 的指数更大
        else {
            Pc.elem[k] = Pb.elem[j];
            k++; 
            j++;
        }
    }

    // 2. 打扫战场：如果 Pa 还有剩余，全抄过来
    while (i < Pa.noelem) {
        Pc.elem[k++] = Pa.elem[i++];
    }
    
    // 3. 打扫战场：如果 Pb 还有剩余，全抄过来
    while (j < Pb.noelem) {
        Pc.elem[k++] = Pb.elem[j++];
    }

    Pc.noelem = k; // 更新结果的总项数

    // 4. 收尾工作：将结果赋给 Pa，并销毁 Pb
    Pa = Pc;       // C/C++ 中，相同类型的结构体可以直接赋值拷贝
    Pb.noelem = 0; // 逻辑上销毁 Pb（将项数置为 0 即可）
}
```

### 第二题：两个有序表的并集 (动态顺序表)

由于要求解并集并去重，如果在原表上直接进行插入操作，会导致大量元素的搬移（时间复杂度骤增）。标准且高效的做法是**创建一个新的顺序表 L3** 来存放结果。

C++

```c++
#include <iostream>
#include <cstdlib> // 包含 malloc 和 free
using namespace std;

#define LIST_INIT_SIZE 100
#define LISTINCREMENT 10

typedef struct {
    int *elem;      // 动态分配数组的基地址
    int length;     // 当前表内的实际元素个数
    int listsize;   // 当前分配的物理容量
} SqList;

// 辅助函数：初始化动态顺序表
void InitList(SqList &L, int capacity = LIST_INIT_SIZE) {
    L.elem = (int *)malloc(capacity * sizeof(int));
    if (!L.elem) {
        cout << "内存分配失败！" << endl;
        exit(1);
    }
    L.length = 0;
    L.listsize = capacity;
}

// 核心函数：求有序表 L1 和 L2 的并集，结果存入 L3
void UnionSqList(SqList L1, SqList L2, SqList &L3) {
    // 优化：并集的最大长度不会超过 L1 和 L2 的长度之和
    // 我们直接为 L3 分配足够的初始空间，避免在合并过程中频繁触发扩容
    InitList(L3, L1.length + L2.length);

    int i = 0; // L1 的游标
    int j = 0; // L2 的游标
    int k = 0; // L3 的游标

    // 1. 双指针比较
    while (i < L1.length && j < L2.length) {
        if (L1.elem[i] < L2.elem[j]) {
            L3.elem[k++] = L1.elem[i++];
        } 
        else if (L1.elem[i] > L2.elem[j]) {
            L3.elem[k++] = L2.elem[j++];
        } 
        // 关键点：当两者相等时，说明是重复元素
        else { 
            L3.elem[k++] = L1.elem[i]; // 只存入一次
            i++; 
            j++; // 两个指针同时往后走，实现去重
        }
    }

    // 2. 打扫战场：把没遍历完的那个表剩下的元素全抄进来
    while (i < L1.length) {
        L3.elem[k++] = L1.elem[i++];
    }
    while (j < L2.length) {
        L3.elem[k++] = L2.elem[j++];
    }

    // 3. 更新 L3 的真实长度
    L3.length = k; 
}
```

这两段代码非常典型地展示了**基于数组结构**进行双指针操作的套路。你可以把它们复制到你的 IDE 里，自己写个 `main` 函数喂点测试数据跑跑看，观察一下游标是怎么移动的，这样理解会更深。
